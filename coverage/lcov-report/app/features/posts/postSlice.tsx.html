export const extendedApiSlice = apiSlice.injectEndpoints({
  endpoints: builder => ({
    getPosts: builder.query<EntityState<Post>, void>({
      query: () => '/posts',
      transformResponse: (responseData: Post[], meta) => {
        let min = 1;
        const loadedPosts = responseData.map(post => {
          if(!post?.date) post.date = sub(new Date(), { minutes: min++ }).toISOString()
          if(!post?.reactions) post.reactions = {
            thumbsUp: 0,
            hooray: 0,
            heart: 0,
            rocket: 0,
            coffee: 0,
          }
 
          return post;
        });
        return postsAdapter.setAll(initialState, loadedPosts)
      },
      providesTags: ( result ) => 
        result
          ? [
              ...result.ids.map(id => ({ type: 'Post' as const, id })),
              { type: 'Post', id: 'LIST' },
            ]
          : [{ type: 'Post', id: 'LIST' }],
      
    }),
    getPostByUserId: builder.query<EntityState<Post>, number>({
      query: id => `/posts/?userId=${id}`,
      transformResponse: (responseData: Post[]) => {
        let min = 1
        const loadedPosts = responseData.map(post => {
          if(!post?.date) post.date = sub(new Date(), { minutes: min++ }).toISOString()
          if(!post?.reactions) post.reactions = {
            thumbsUp: 0,
            hooray: 0,
            heart: 0,
            rocket: 0,
            coffee: 0,
          }
 
          return post;
        })
        return postsAdapter.setAll(initialState, loadedPosts)
      },
      providesTags: ( result ) => 
        result
          ? [
              ...result.ids.map(id => ({ type: 'Post' as const, id })),
              { type: 'Post', id: 'LIST' },
            ]
          : [{ type: 'Post', id: 'LIST' }],
    }),

    }),
    updatePost: builder.mutation({

        body: {
          ...initialPost,

        }
      }),
      invalidatesTags: (result, error, arg) => [
        { type: 'Post' as const, id: arg.id }
      ]
    }),
    deletePost: builder.mutation({
      query: ({ id }) => ({
        url: `/posts/${id}`,
        method: 'DELETE',
        body: { id }
      }),
      invalidatesTags: (result, error, arg) => [
        { type: 'Post', id: arg.id }
      ]
    }),
    addReaction: builder.mutation({
      query: ({ postId, reactions }) => ({
        url: `posts/${postId}`,
        method: 'PATCH',
        body: { reactions }
      }),
      async onQueryStarted({ postId, reactions }, {dispatch, queryFulfilled}) {
        
        const patchResult = dispatch(

            if (post) post.reactions = reactions

        try {
          await queryFulfilled

          patchResult.undo()
        }
